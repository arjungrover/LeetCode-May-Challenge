/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */

//we can have static variable /heap variable and decrement in inorder(Sorted)
class Solution {
    public int kthSmallest(TreeNode root, int k) {
        Stack<TreeNode> stack = new Stack<>();
        
        while(true){
            //preorder
            while(root!=null){
                stack.push(root);
                root = root.left;
            }
            root = stack.pop();   
            if(--k==0) return root.val;  //inorder
            root = root.right;   //post order
        }
    }
}
