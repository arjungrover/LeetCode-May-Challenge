
//Time complexity O(log n)
//space complexity O(1)


//for handling test case 2147483647 which is INTEGER.MAX_VALUE , We have to take long .
//for num=0 , ans is false given
class Solution {
    public boolean isPerfectSquare(int num) {
        if(num<1)
            return true;
        
        long left = 1;
        long right = num;
        while(left<=right){
            long mid = (left+right)/2;
            if(mid*mid==num)
                return true;
            else if(mid*mid<num)
                left = mid+1;
            else
                //when sqrt is greater than number
                right = mid-1;
            
            // System.out.println(mid);
               
        }
        return false;
    }
}
